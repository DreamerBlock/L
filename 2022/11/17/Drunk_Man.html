import random
x = 0
y = 0
nights = 0
turn = 0
turns = []

while (nights < 10):
    step = random.randrange(4)
    if step == 0:
        x = x+1
    if step == 1:
        x = x-1
    if step == 2:
        y = y+1
    if step == 3:
        y = y-1
    
    turn = turn + 1
    
    if x == 0 and y == 0:
        nights = nights + 1
        print("The Man Has Made It Home After ", turn, "Turns")
        turns.append(turn)
        turn = 0
    if turn/1000 % 1000 == 0:
        print("(", x,y, ")")

average = sum(turns) / len(turns)
print(average)
